178  SbBool isSpace(const char c) {
179  // For vrml97, comma is treated as whitespace. Added this function
180  // and converted isspace() calls to calls to this function.
181  // 2001-10-24, pederb.
182  //
183  // From the VRML97 specification:
184  // The carriage return (0x0d), linefeed (0x0a), space (0x20), tab
185  // (0x09), and comma (0x2c) characters are whitespace characters
186  // wherever they appear outside of quoted SFString or MFString
187  // fields. Any number of whitespace characters and comments may be
188  // used to separate the syntactic entities of a VRML file.
189  //
190  //
192  //
193  // ANSI C isspace() takes the current locale into account. Under
194  // Microsoft Windows, this can lead to "interesting" artifacts, like a
195  // case with RR tracked down and fixed by <thammer@sim.no> where a
196  // character (was it Ã¼?) with ASCII value > 127 made isspace()
197  // return non-nil on a German system. So we're using our own
198  // locale-independent isspace() implementation instead.
199  return coin_isspace(c) || (this->vrml2file && c == ',');
200  }
