48 /* Note: Under Win32, we need to make sure we use the correct calling
49  method by using the OPENALWRAPPER_APIENTRY define for the function
50  signature types (or else we'll get weird stack errors). The calling
51  convention must match the calling convention used in the
52  OpenAL32.dll we link against. Both the LGPL version of OpenAL in
53  CVS (www.openal.org), Creative Lab's (binary) SDK, and nVidia's
54  AudioSDK, uses this calling method, so I think it is quite safe to
55  assume that all OpenAL32.dlls use this method. On other platforms,
56  just define OPENALWRAPPER_APIENTRY empty. Stack errors (under Win32)
57  can be detected by specifying the /GZ linker option.
58  2003-03-19 thammer. */
59 
60 #ifdef _WIN32
61  #define OPENALWRAPPER_APIENTRY __cdecl
62 #else
63  #define OPENALWRAPPER_APIENTRY
64 #endif /* _WIN32 */
