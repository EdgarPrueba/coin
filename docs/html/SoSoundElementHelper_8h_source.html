73  void traverseInactiveChildren(SoNode *node, SoAction *action, int idx,
74  SoAction::PathCode pathcode,
75  int numchildren,
76  SoChildList *children) {
77  if (this->shouldDoSoundTraversal(action, idx, pathcode)) {
78  // Note: If there is a playing SoVRMLSound node somewhere among
79  // the non-active children sub-graphs, it must be informed that
80  // it shouldn't be playing anymore. So we traverse all inactive
81  // children. This could be optimized, as we only need to visit
82  // the old active (now inactive) child. 2003-02-04 thammer.
83 
84  SoState * state = action->getState();
85  int n = numchildren;
86  for (int i=0; i<n; i++) {
87  if (i != idx) {
88  action->getState()->push();
89  SoSoundElement::setIsPartOfActiveSceneGraph(state, node, FALSE);
90  children->traverse(action, i);
91  action->getState()->pop();
92  }
93  }
94  this->shoulddosoundtraversal = FALSE;
95  }
96  }

98  void preAudioRender(SoNode *node, SoAction *action) {
99  this->lookforsoundnode = FALSE;
100  this->oldhassound = FALSE;
101  this->oldisplaying = FALSE;
102 
103  int numindices;
104  const int * indices;
105  SoState * state = action->getState();
106  if ((action->getPathCode(numindices, indices) != SoAction::IN_PATH) &&
107  (this->hassoundchild != SoSoundElementHelper::NO) ) {
108  this->lookforsoundnode = TRUE;
109  this->oldhassound = SoSoundElement::setSceneGraphHasSoundNode(state,
110  node, FALSE);
111  this->oldisplaying = SoSoundElement::setSoundNodeIsPlaying(state,
112  node, FALSE);
113  }
114  }
115 
116  void postAudioRender(SoNode *node, SoAction *action) {
117  if (this->lookforsoundnode) {
118  SoState * state = action->getState();
119  SbBool soundnodefound = SoSoundElement::sceneGraphHasSoundNode(state);
120  this->soundchildplaying = SoSoundElement::soundNodeIsPlaying(state);
121  SoSoundElement::setSceneGraphHasSoundNode(state, node, this->oldhassound
122  || soundnodefound);
123  SoSoundElement::setSoundNodeIsPlaying(state, node, this->oldisplaying
124  || this->soundchildplaying);
125  this->hassoundchild = soundnodefound ? SoSoundElementHelper::YES :
126  SoSoundElementHelper::NO;
127  }
128  }

133  return action->isOfType(SoAudioRenderAction::getClassTypeId()) &&
134  (this->hassoundchild != SoSoundElementHelper::NO) &&
135  (! ( (idx >= 0) && (pathcode == SoAction::IN_PATH) ) ) &&
136  this->shoulddosoundtraversal;
137  }
